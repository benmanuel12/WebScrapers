For each new product
if no card exists in cards table with that name
    create a new CardAnnotation from name, image and set code (if availible)
    push to database
if no option exists in option table from that site with that cardID
    create a new OptionAnnotation from cardID, image, price and link
    push to database
else (card already in database and option also already in database)
    fetch that option and update it's price
